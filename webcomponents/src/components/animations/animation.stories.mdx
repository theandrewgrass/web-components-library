import { html } from 'lit-html';
import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs';
import 'Components/animations/animation';
import 'Components/animations/checkmark';
import { BasicSpinner, GradientSpinner } from 'Components/animations/spinners';

<Meta
  title='Animations'
  component='kobo-animation'
  args = {{
    containerSize: 50,
  }}
  argTypes={{
    containerSize: {
      name: 'Container Size (px)',
      control: {
        type: 'number',
        min: 0,
        max: 100,
        step: 10,
      },
    },
  }}
/>

export const Template = (args) => {
  return html`
    <kobo-animation 
      .animationWidth=${args.containerSize} 
      .animationHeight=${args.containerSize}
    >
      ${args.animation}
    </kobo-animation>
  `;
}

export const SpinnerTemplate = (args) => {
  return html`
    <kobo-animation 
      .animationWidth=${args.containerSize} 
      .animationHeight=${args.containerSize}
    >
      <basic-spinner slot="animation" .spinnerType=${args.spinnerType}></basic-spinner>
    </kobo-animation>
  `;
}

# Basic Spinner

A basic spinner to be used to represent an action currently in progress. Grows to the width of the element within which it's contained.
